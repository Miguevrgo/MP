108 void quickSort(BigramFreq array[], int nElements, int start, int end){
109  if (start >= end) {
110  //In this case the array is sorted
111  return;
112  }
113 
114  //Choosing the pivot
115  int pivotIndex = start + (end - start) / 2;
116  BigramFreq pivot = array[pivotIndex];
117 
118  //We divide the array around the pivot element
119  int i = start;
120  int j = end;
121  while (i <= j) {
122  while (array[i].getFrequency() > pivot.getFrequency() ||
123  (array[i].getFrequency() == pivot.getFrequency() &&
124  array[i].getBigram().toString() < pivot.getBigram().toString())) {
125  i++;
126  }
127  while (array[j].getFrequency() < pivot.getFrequency() ||
128  (array[j].getFrequency() == pivot.getFrequency() &&
129  array[j].getBigram().toString() > pivot.getBigram().toString())) {
130  j--;
131  }
132  if (i <= j) {
133  swapElementsArrayBigramFreq(array, nElements, i, j);
134  i++;
135  j--;
136  }
137  }
138 
139  //Sorting the two subarrays using a recursive method
140  if (start < j) {
141  quickSort(array, nElements, start, j);
142  }
143  if (i < end) {
144  quickSort(array, nElements, i, end);
145  }
